Google Charts

Google chart tools are powerful, simple to use, and free. Try out our rich gallery of interactive charts and data tools. Google chart wraps with angular google chart for angularjs app.

Area Chart


<div chart="googleAreaChart" style="width: 100%; height: 400px;" google-chart></div>
                            

$scope.googleAreaChart = {
    "type": "AreaChart",
    "displayed": false,
    "data": [
        ['Year', 'Sales', 'Expenses'],
        ['2013', 1000, 400],
        ['2014', 1170, 460],
        ['2015', 660, 1120],
        ['2016', 1030, 540]
    ],
    "options": {
        title: 'Company Performance',
        hAxis: { title: 'Year', titleTextStyle: { color: '#333' } },
        vAxis: { minValue: 0 }
    }
}

Line Chart


<div chart="googleLineChart" style="width: 100%; height: 400px;" google-chart></div>
                            

$scope.googleLineChart = {
    "type": "LineChart",
    "displayed": false,
    "data": [
        ['Year', 'Sales', 'Expenses'],
        ['2004', 1000, 400],
        ['2005', 1170, 460],
        ['2006', 660, 1120],
        ['2007', 1030, 540]
    ],
    "options": {
        chart: {
            title: 'Average Temperatures and Daylight in Iceland Throughout the Year'
        },
        series: {
            // Gives each series an axis name that matches the Y-axis below.
            0: { axis: 'Temps' },
            1: { axis: 'Daylight' }
        },
        axes: {
            // Adds labels to each axis; they don't have to match the axis names.
            y: {
                Temps: { label: 'Temps (Celsius)' },
                Daylight: { label: 'Daylight' }
            }
        }
    },
    "formatters": {}
}

Gauge Chart


<div chart="googleGaugeChart" google-chart></div>
                            

$scope.googleGaugeChart = {
    "type": "Gauge",
    "data": [
        ['Label', 'Value'],
        ['Memory', 80],
        ['CPU', 55],
        ['Network', 68]
    ],
    "options": {
        width: 400,
        height: 120,
        redFrom: 90,
        redTo: 100,
        yellowFrom: 75,
        yellowTo: 90,
        minorTicks: 5
    }
}
$interval(function() {
    $scope.googleGaugeChart.data[1][1] = Math.round(60 * Math.random());
    $scope.googleGaugeChart.data[2][1] = Math.round(60 * Math.random());
    $scope.googleGaugeChart.data[3][1] = Math.round(60 * Math.random());
}, 1000);
