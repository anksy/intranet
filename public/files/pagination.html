Pagination

Add pagination links to help split up your long content into shorter, easier to understand blocks.

Basic Usage

Loaded data from page {{currentPage}}

In HTML sample code angularjs scope variable denoted as {} instead of {{ }}. 

<!-- Loaded content -->
<div layout="row" layout-align="center center">
    <p>Loaded data from page<strong> {currentPage} </strong></p>
</div>
<!--End loaded content -->
<!-- paging wrapper -->
<div class="paging-wrapper">
    <cl-paging cl-pages="paging.total" cl-steps="6" cl-page-changed="paging.onPageChanged()" cl-align="center center" cl-current-page="paging.current"></cl-paging>
</div>
<!-- end paging wrapper -->
                            

$scope.currentPage = 0;
$scope.paging = {
    total: 100,
    current: 1,
    onPageChanged: loadPages
};

function loadPages() {
    // TODO : Load current page Data here
    $scope.currentPage = $scope.paging.current;
}
