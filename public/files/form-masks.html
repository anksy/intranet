Form Masks

Cleave.js has a simple purpose: to help you format input text content automatically.


<form name="formMaskForm" class="row">
    <md-input-container class="col s12 l6 md-block">
        <label>Date</label>
        <input type="text" name="date" placeholder="2017-01-01" cleave="formMaskOptions.date" ng-model="model.date" />
    </md-input-container>

    <md-input-container class="col s12 l6 md-block">
        <label>Date</label>
        <input type="text" name="date1" placeholder="01/01/2017" cleave="formMaskOptions.date1" ng-model="model.date1">
    </md-input-container>

    <md-input-container class="col s12 l6 md-block">
        <label>Credit Card</label>
        <input type="text" name="creditCard" placeholder="XXXX-XXXXX-XXXXXX" cleave="formMaskOptions.creditCard" ng-model="model.creditCard">
    </md-input-container>

    <md-input-container class="col s12 l6 md-block">
        <label>Purchase Code</label>
        <input type="text" name="purchaseCode" cleave="formMaskOptions.purchaseCode" ng-model="model.purchaseCode">
    </md-input-container>

    <md-input-container class="col s12 l6 md-block">
        <label>Currency</label>
        <input type="text" name="currency" cleave="formMaskOptions.currency" ng-model="model.currency">
    </md-input-container>

    <md-input-container class="col s12 l6 md-block">
        <label>Mobile Number</label>
        <input type="text" name="mobile" placeholder="1234567891" cleave="formMaskOptions.mobile" ng-model="model.mobile">
    </md-input-container>
</form>
                            

$scope.formMaskOptions = {
    date: {
        date: true,
        datePattern: ['Y', 'm', 'd'],
        delimiters: ['-', '-', '-']
    },
    date1: {
        date: true,
        datePattern: ['m', 'd', 'Y']
    },
    creditCard: {
        creditCard: true,
        delimiter: '-'
    },
    purchaseCode: {
        blocks: [6, 3, 3, 3],
        prefix: 'ISBN',
        uppercase: true,
        delimiters: ['-', '-', '-', '-']
    },
    currency: {
        numeral: true,
        prefix: '$ ',
        rawValueTrimPrefix: true
    },
    mobile: {
        blocks: [10]
    }
};

$scope.model = {
    date: '',
    date1: '',
    creditCard: '',
    purchaseCode: 'ISBN',
    currency: '$ ',
    mobile: ''
};
