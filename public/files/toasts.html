$scope.simpleToast = function(toastPosition) {
    $mdToast.show(
        $mdToast.simple()
        .textContent('Simple Toast!') // toast content
        .position(toastPosition) // toast position
        .hideDelay(3000) // toast duration
    );
};

Toast with Action

Show With Action 

<md-button class="md-raised md-primary" ng-click="actionToast()">Show With Action</md-button>
                            

$scope.actionToast = function() {
    var toast = $mdToast.simple()
        .textContent('Marked as read')
        .action('UNDO')
        .highlightAction(true)
        .highlightClass('md-warn') // warn is used by default, this just demonstrates the usage.
        .position('top');

    $mdToast.show(toast).then(function(response) {
        if (response == 'ok') {
            alert('You clicked the \'UNDO\' action.');
        }
    });
};

Custom Usage

Show Custom Toast 

<md-button class="md-raised md-primary" ng-click="showCustomToast()">Show Custom Toast</md-button>
                            

$scope.showCustomToast = function() {
    $mdToast.show({
        hideDelay: 3000,
        position: 'top right',
        template: '<md-toast> <span class="md-toast-text" flex>Custom toast!</span> <md-button class="md-warn"> More info </md-button> <md-button> Close </md-button> </md-toast>'
    });
};
