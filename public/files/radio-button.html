Radio Buttons

Radio Buttons are used when the user must make only one selection out of a group of items.

Basic Usage

Apple Banana Mango Orange(Disabled) Lemon(No Ink)

Selected Value: {{ basicUsageRadioButton.group1 }}

In HTML sample code angularjs scope variable denoted as {} instead of {{ }}. 

<form>
    <md-radio-group ng-model="basicUsageRadioButton.group1">
        <md-radio-button value="Apple" class="md-primary">Apple</md-radio-button>
        <md-radio-button value="Banana" class="md-warn"> Banana </md-radio-button>
        <md-radio-button value="Mango">Mango</md-radio-button>
        <md-radio-button value="Orange" ng-disabled="true">Orange(Disabled)</md-radio-button>
        <md-radio-button value="Lemon" class="md-primary" md-no-ink>Lemon(No Ink)</md-radio-button>
    </md-radio-group>
    <p>Selected Value: <span class="red">{ basicUsageRadioButton.group1 }</span> </p>
</form>
                            

$scope.basicUsageRadioButton = {
    group1: 'Banana'
};

Color Options

primary warn accent purple blue
cyan teal green yellow brown 

<md-radio-group ng-model="colorOptionRadioButton.group">
    <md-radio-button value="purple" class="md-purple">Purple</md-radio-button>
</md-radio-group>
                            

$scope.colorOptionRadioButton = {
    group: 'primary'
};


md-radio-button:not([disabled]).md-purple .md-on,
md-radio-group:not([disabled]) .md-purple .md-on {
    background-color: #9C27B0;
}

md-radio-button:not([disabled]).md-purple.md-checked .md-off,
md-radio-group:not([disabled]) .md-purple.md-checked .md-off {
    border-color: #9C27B0;
}
